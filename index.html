#include using namespace std; #define MAX 5 // Queue size class Queue { int arr[MAX]; int front, rear; public: // Constructor Queue() { front = 0; rear = -1; } // Enqueue operation void enqueue(int value) { if (rear == MAX - 1) { cout << "Queue Overflow\n"; } else { rear++; arr[rear] = value; cout << value << " inserted into queue\n"; } } // Dequeue operation void dequeue() { if (front > rear) { cout << "Queue Underflow\n"; } else { cout << arr[front] << " removed from queue\n"; front++; } } // Peek operation void peek() { if (front > rear) { cout << "Queue is empty\n"; } else { cout << "Front element: " << arr[front] << endl; } } // Display queue void display() { if (front > rear) { cout << "Queue is empty\n"; } else { cout << "Queue elements: "; for (int i = front; i <= rear; i++) { cout << arr[i] << " "; } cout << endl; } } }; // Main function int main() { Queue q; q.enqueue(10); q.enqueue(20); q.enqueue(30); q.display(); q.peek(); q.dequeue(); q.display(); return 0; }
